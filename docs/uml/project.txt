@startuml
start
:Create layout;

if (venv_path not empty) then (yes)
  :Create venv on venv_path;
else (no)
  if (venv?) then (yes)
    :Create venv on PATH;
    :Set venv_path = PATH;
  else (no)
  endif
endif

if (direnv = True and venv_path not empty?) then (yes)
  :Setup direnv;
else (no)
endif
end
@enduml
